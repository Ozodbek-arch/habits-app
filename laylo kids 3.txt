import React, { useState, useEffect } from 'react';
import { createClient } from '@supabase/supabase-js';

// 1. Supabase-ni sozlash (O'zingizning URL va KEY laringizni kiriting)
const SUPABASE_URL = "https://your-supabase-url.supabase.co";
const SUPABASE_KEY = "your-anon-key";
const supabase = createClient(SUPABASE_URL, SUPABASE_KEY);

// Telegram bot sozlamalari (O'zingizning BOT_TOKEN va CHAT_ID ni yozasiz)
const TELEGRAM_BOT_TOKEN = "YOUR_BOT_TOKEN";
const TELEGRAM_CHAT_ID = "YOUR_CHAT_ID"; // Shaxsiy ID yoki Kanal/Guruh ID-si

export default function App() {
  const [lang, setLang] = useState('uz');
  const t = translations[lang];

  // State-lar
  const [categories, setCategories] = useState([]);
  const [products, setProducts] = useState([]);
  const [cart, setCart] = useState([]);
  const [selectedCategory, setSelectedCategory] = useState('all');
  const [searchQuery, setSearchQuery] = useState('');
  
  // Buyurtma modal state-lari
  const [isCheckoutOpen, setIsCheckoutOpen] = useState(false);
  const [customerName, setCustomerName] = useState('');
  const [customerPhone, setCustomerPhone] = useState('');
  const [customerRegion, setCustomerRegion] = useState('');
  const [deliveryType, setDeliveryType] = useState('pickup'); // pickup yoki delivery

  // Admin state-lari
  const [isAdmin, setIsAdmin] = useState(false);
  const [showCategoryModal, setShowCategoryModal] = useState(false);
  const [editingCategory, setEditingCategory] = useState(null);
  const [newCategoryName, setNewCategoryName] = useState('');
  
  const [imageFile, setImageFile] = useState(null); // Supabase storage uchun rasm fayli

  // Yuklash (Fetch) funksiyalari
  useEffect(() => {
    fetchCategories();
    fetchProducts();
  }, []);

  const fetchCategories = async () => {
    const { data, error } = await supabase.from('categories').select('*');
    if (!error) setCategories(data);
  };

  const fetchProducts = async () => {
    const { data, error } = await supabase.from('products').select('*');
    if (!error) setProducts(data);
  };

  // --- KATALOG (KATEGORIYA) BOSHQARUVI ---
  const handleSaveCategory = async () => {
    if (!newCategoryName) return;

    if (editingCategory) {
      // Tahrirlash
      await supabase
        .from('categories')
        .update({ name: newCategoryName })
        .eq('id', editingCategory.id);
    } else {
      // Yangi qo'shish
      await supabase
        .from('categories')
        .insert([{ name: newCategoryName }]);
    }

    setNewCategoryName('');
    setEditingCategory(null);
    setShowCategoryModal(false);
    fetchCategories();
  };

  const handleDeleteCategory = async (id) => {
    if (window.confirm("Katalogni o'chirmoqchimisiz? Ichidagi mahsulotlar yetim qolishi mumkin.")) {
      await supabase.from('categories').delete().eq('id', id);
      fetchCategories();
    }
  };

  // --- SUPABASE STORAGE GA RASM YUKLASH (IMAGE UPLOAD) ---
  const handleImageUpload = async (file) => {
    if (!file) return null;
    const fileExt = file.name.split('.').pop();
    const fileName = `${Math.random()}.${fileExt}`;
    const filePath = `products/${fileName}`;

    // 'images' nomli bucket-ga rasm yuklash
    const { error: uploadError } = await supabase.storage
      .from('images')
      .upload(filePath, file);

    if (uploadError) {
      console.error(uploadError);
      return null;
    }

    // Yuklangan rasmning URL manzilini olish
    const { data } = supabase.storage.from('images').getPublicUrl(filePath);
    return data.publicUrl;
  };

  // --- DOSTAVKA NARXINI HISOBLASH ---
  const getDeliveryCost = () => {
    return deliveryType === 'delivery' ? 30000 : 0; // Yetkazib berish bo'lsa +30,000 so'm
  };

  const getCartSubtotal = () => {
    return cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);
  };

  const getCartTotal = () => {
    return getCartSubtotal() + getDeliveryCost();
  };

  // --- TELEGRAM BOT INTEGRATSIYASI VA BUYURTMA ---
  const handlePlaceOrder = async () => {
    if (!customerName || !customerPhone || !customerRegion) {
      alert("Iltimos barcha maydonlarni to'ldiring!");
      return;
    }

    const subtotal = getCartSubtotal();
    const deliveryCost = getDeliveryCost();
    const total = getCartTotal();

    // 1. Supabase-ga buyurtmani saqlash
    const { data: orderData, error } = await supabase.from('orders').insert([
      {
        customer_name: customerName,
        phone: customerPhone,
        region: customerRegion,
        delivery_type: deliveryType,
        delivery_cost: deliveryCost,
        total_price: total,
        items: cart
      }
    ]).select();

    if (error) {
      alert("Xatolik yuz berdi. Qaytadan urinib ko'ring.");
      return;
    }

    // 2. Telegram guruhga chiroyli xabar yuborish
    let orderItemsText = cart.map(item => `👕 ${item.name} x ${item.quantity} - ${(item.price * item.quantity).toLocaleString()} so'm`).join('\n');
    
    const telegramMessage = `
🛍 **YANGI BUYURTMA (Gulnora Shop)**
━━━━━━━━━━━━━━━━━━
👤 **Mijoz:** ${customerName}
📞 **Telefon:** ${customerPhone}
📍 **Viloyat/Shahar:** ${customerRegion}
🚚 **Dostavka:** ${deliveryType === 'delivery' ? "Yetkazib berish (30,000 so'm)" : "Olib ketish (Bepul)"}

📦 **Mahsulotlar:**
${orderItemsText}

💵 **Tovarlar narxi:** ${subtotal.toLocaleString()} so'm
🚚 **Yetkazib berish:** ${deliveryCost.toLocaleString()} so'm
💰 **JAMI TO'LOV:** ${total.toLocaleString()} so'm
━━━━━━━━━━━━━━━━━━
    `;

    try {
      await fetch(`https://api.telegram.org/bot${TELEGRAM_BOT_TOKEN}/sendMessage`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          chat_id: TELEGRAM_CHAT_ID,
          text: telegramMessage,
          parse_mode: "Markdown"
        })
      });
    } catch (tgError) {
      console.error("Telegramga yuborishda xatolik:", tgError);
    }

    alert("Rahmat! Buyurtmangiz qabul qilindi.");
    setCart([]);
    setIsCheckoutOpen(false);
  };

  return (
    <div className="min-h-screen bg-gray-50 text-gray-800 pb-12" dir={lang === 'ar' ? 'rtl' : 'ltr'}>
      {/* --- HEADER --- */}
      <header className="bg-white shadow-sm sticky top-0 z-40">
        <div className="max-w-6xl mx-auto px-4 py-4 flex flex-wrap justify-between items-center gap-4">
          <div>
            <h1 className="text-3xl font-extrabold text-pink-600">{t.title}</h1>
            <p className="text-sm text-gray-500">{t.subtitle}</p>
          </div>

          <div className="flex items-center gap-4">
            {/* Tillar ro'yxati dropdowni */}
            <div className="flex items-center gap-2">
              <span className="text-xs text-gray-400 font-semibold">{t.selectLang}:</span>
              <select 
                value={lang} 
                onChange={(e) => setLang(e.target.value)}
                className="bg-gray-100 border border-gray-300 rounded px-2 py-1 text-sm font-medium outline-none focus:ring-2 focus:ring-pink-500"
              >
                <option value="uz">🇺🇿 O'zbekcha</option>
                <option value="ru">🇷🇺 Русский</option>
                <option value="en">🇬🇧 English</option>
                <option value="ar">🇸🇦 العربية (RTL)</option>
                <option value="zh">🇨🇳 汉语</option>
              </select>
            </div>

            {/* Admin rejimi tugmasi */}
            <button 
              onClick={() => setIsAdmin(!isAdmin)}
              className={`px-4 py-1.5 rounded-lg text-sm font-semibold transition ${isAdmin ? 'bg-pink-600 text-white' : 'bg-gray-100 hover:bg-gray-200'}`}
            >
              {isAdmin ? "User Panel" : "Admin Panel"}
            </button>
          </div>
        </div>
      </header>

      {/* --- ASOSIY QISM --- */}
      <main className="max-w-6xl mx-auto px-4 mt-6">
        
        {/* ADMIN BO'LIMI: KATALOGLARNI TARXIRLASH */}
        {isAdmin && (
          <div className="bg-white p-6 rounded-2xl shadow-md border border-pink-100 mb-8">
            <div className="flex justify-between items-center mb-4">
              <h2 className="text-xl font-bold text-gray-700">{t.categories} (Katalog boshqaruvi)</h2>
              <button 
                onClick={() => { setEditingCategory(null); setNewCategoryName(''); setShowCategoryModal(true); }}
                className="bg-pink-600 hover:bg-pink-700 text-white px-4 py-2 rounded-lg text-sm font-semibold"
              >
                + {t.addCategory}
              </button>
            </div>

            {/* Kataloglar ro'yxati tahrirlash paneli */}
            <div className="grid grid-cols-2 sm:grid-cols-3 md:grid-cols-4 gap-4">
              {categories.map(cat => (
                <div key={cat.id} className="p-3 bg-gray-50 border rounded-xl flex items-center justify-between">
                  <span className="font-semibold">{cat.name}</span>
                  <div className="flex gap-1">
                    <button 
                      onClick={() => { setEditingCategory(cat); setNewCategoryName(cat.name); setShowCategoryModal(true); }}
                      className="text-blue-500 hover:text-blue-700 text-xs px-2"
                    >
                      ✏️
                    </button>
                    <button 
                      onClick={() => handleDeleteCategory(cat.id)}
                      className="text-red-500 hover:text-red-700 text-xs px-2"
                    >
                      🗑
                    </button>
                  </div>
                </div>
              ))}
            </div>
          </div>
        )}

        {/* --- MAHSULOTLAR VA SARIQLASH (FILTR) --- */}
        {!isAdmin && (
          <div className="flex flex-col md:flex-row gap-6">
            {/* Katalog menyusi */}
            <div className="w-full md:w-1/4">
              <div className="bg-white p-4 rounded-2xl shadow-sm border sticky top-24">
                <h3 className="font-bold text-lg mb-3">{t.categories}</h3>
                <div className="flex flex-row md:flex-col gap-2 overflow-x-auto pb-2 md:pb-0">
                  <button 
                    onClick={() => setSelectedCategory('all')}
                    className={`whitespace-nowrap px-4 py-2 rounded-lg text-left text-sm font-semibold ${selectedCategory === 'all' ? 'bg-pink-500 text-white' : 'hover:bg-gray-100'}`}
                  >
                    {t.all}
                  </button>
                  {categories.map(cat => (
                    <button 
                      key={cat.id}
                      onClick={() => setSelectedCategory(cat.id)}
                      className={`whitespace-nowrap px-4 py-2 rounded-lg text-left text-sm font-semibold ${selectedCategory === cat.id ? 'bg-pink-500 text-white' : 'hover:bg-gray-100'}`}
                    >
                      {cat.name}
                    </button>
                  ))}
                </div>
              </div>
            </div>

            {/* Maxsulotlar ro'yxati */}
            <div className="w-full md:w-3/4">
              <div className="grid grid-cols-1 sm:grid-cols-2 md:grid-cols-3 gap-6">
                {products
                  .filter(p => selectedCategory === 'all' || p.category_id === selectedCategory)
                  .map(product => (
                    <div key={product.id} className="bg-white rounded-2xl shadow-sm hover:shadow-md transition border overflow-hidden">
                      <img src={product.image || 'placeholder.jpg'} alt={product.name} className="h-64 w-full object-cover" />
                      <div className="p-4">
                        <h4 className="font-bold text-lg text-gray-800">{product.name}</h4>
                        <p className="text-pink-600 font-extrabold mt-2 text-xl">{product.price.toLocaleString()} so'm</p>
                        <button 
                          onClick={() => {
                            const found = cart.find(item => item.id === product.id);
                            if (found) {
                              setCart(cart.map(item => item.id === product.id ? {...item, quantity: item.quantity + 1} : item));
                            } else {
                              setCart([...cart, {...product, quantity: 1}]);
                            }
                          }}
                          className="w-full bg-pink-500 hover:bg-pink-600 text-white py-2 rounded-xl mt-4 font-semibold transition"
                        >
                          + {t.addToCart}
                        </button>
                      </div>
                    </div>
                ))}
              </div>
            </div>
          </div>
        )}
      </main>

      {/* --- KATALOG QO'SHISH/TAHRIRLASH MODAL OYNASI --- */}
      {showCategoryModal && (
        <div className="fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center p-4 z-50">
          <div className="bg-white p-6 rounded-2xl shadow-xl w-full max-w-md">
            <h3 className="text-xl font-bold mb-4">{editingCategory ? t.editCategory : t.addCategory}</h3>
            <div className="mb-4">
              <label className="block text-sm font-semibold mb-2">{t.categoryName}</label>
              <input 
                type="text" 
                value={newCategoryName} 
                onChange={(e) => setNewCategoryName(e.target.value)}
                placeholder="Masalan: Qizlar kiyimi" 
                className="w-full border rounded-lg px-3 py-2 outline-none focus:ring-2 focus:ring-pink-500" 
              />
            </div>
            <div className="flex justify-end gap-2">
              <button 
                onClick={() => setShowCategoryModal(false)}
                className="px-4 py-2 bg-gray-200 rounded-lg text-sm font-semibold"
              >
                Bekor qilish
              </button>
              <button 
                onClick={handleSaveCategory}
                className="px-4 py-2 bg-pink-600 text-white rounded-lg text-sm font-semibold"
              >
                {t.save}
              </button>
            </div>
          </div>
        </div>
      )}

      {/* --- SAVATCHA VA BUYURTMA MODAL OYNASI --- */}
      {cart.length > 0 && !isAdmin && (
        <div className="fixed bottom-6 right-6 z-50">
          <button 
            onClick={() => setIsCheckoutOpen(true)}
            className="bg-pink-600 hover:bg-pink-700 text-white px-6 py-4 rounded-full shadow-2xl flex items-center gap-3 font-bold text-lg animate-bounce"
          >
            🛒 {t.cart} ({cart.reduce((sum, i) => sum + i.quantity, 0)})
          </button>
        </div>
      )}

      {isCheckoutOpen && (
        <div className="fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center p-4 z-50 overflow-y-auto">
          <div className="bg-white p-6 rounded-3xl shadow-2xl w-full max-w-lg">
            <div className="flex justify-between items-center mb-4">
              <h3 className="text-2xl font-bold text-gray-800">{t.checkout}</h3>
              <button onClick={() => setIsCheckoutOpen(false)} className="text-gray-400 hover:text-gray-600 text-2xl">&times;</button>
            </div>

            {/* Buyurtma elementlari */}
            <div className="max-h-48 overflow-y-auto border-b pb-4 mb-4">
              {cart.map(item => (
                <div key={item.id} className="flex justify-between items-center py-2">
                  <span className="font-semibold">{item.name} (x{item.quantity})</span>
                  <span>{(item.price * item.quantity).toLocaleString()} so'm</span>
                </div>
              ))}
            </div>

            {/* Ma'lumotlarni kiritish oynasi */}
            <div className="space-y-4">
              <div>
                <label className="block text-sm font-semibold mb-1">{t.name}</label>
                <input 
                  type="text" 
                  value={customerName} 
                  onChange={(e) => setCustomerName(e.target.value)}
                  className="w-full border rounded-xl px-3 py-2 outline-none focus:ring-2 focus:ring-pink-500" 
                />
              </div>

              <div>
                <label className="block text-sm font-semibold mb-1">{t.phone}</label>
                <input 
                  type="tel" 
                  value={customerPhone} 
                  onChange={(e) => setCustomerPhone(e.target.value)}
                  className="w-full border rounded-xl px-3 py-2 outline-none focus:ring-2 focus:ring-pink-500" 
                  placeholder="+998 90 123 45 67"
                />
              </div>

              <div>
                <label className="block text-sm font-semibold mb-1">{t.region}</label>
                <select 
                  value={customerRegion} 
                  onChange={(e) => setCustomerRegion(e.target.value)}
                  className="w-full border rounded-xl px-3 py-2 outline-none focus:ring-2 focus:ring-pink-500"
                >
                  <option value="">Viloyatni tanlang</option>
                  <option value="Toshkent sh.">Toshkent sh.</option>
                  <option value="Toshkent vil.">Toshkent vil.</option>
                  <option value="Andijon">Andijon</option>
                  <option value="Farg'ona">Farg'ona</option>
                  <option value="Namangan">Namangan</option>
                  <option value="Buxoro">Buxoro</option>
                  <option value="Samarqand">Samarqand</option>
                  <option value="Qashqadaryo">Qashqadaryo</option>
                  <option value="Xorazm">Xorazm</option>
                </select>
              </div>

              <div>
                <label className="block text-sm font-semibold mb-1">{t.delivery}</label>
                <div className="flex gap-4">
                  <label className="flex items-center gap-2 cursor-pointer">
                    <input 
                      type="radio" 
                      name="delivery" 
                      value="pickup" 
                      checked={deliveryType === 'pickup'} 
                      onChange={() => setDeliveryType('pickup')}
                    />
                    <span>Olib ketish (Bepul)</span>
                  </label>
                  <label className="flex items-center gap-2 cursor-pointer">
                    <input 
                      type="radio" 
                      name="delivery" 
                      value="delivery" 
                      checked={deliveryType === 'delivery'} 
                      onChange={() => setDeliveryType('delivery')}
                    />
                    <span>Dostavka (+30,000 so'm)</span>
                  </label>
                </div>
              </div>
            </div>

            {/* Narx hisob-kitoblari */}
            <div className="bg-gray-50 p-4 rounded-2xl mt-6 space-y-2">
              <div className="flex justify-between text-sm">
                <span>Tovarlar summasi:</span>
                <span>{getCartSubtotal().toLocaleString()} so'm</span>
              </div>
              <div className="flex justify-between text-sm">
                <span>Dostavka:</span>
                <span>{getDeliveryCost().toLocaleString()} so'm</span>
              </div>
              <div className="flex justify-between font-bold text-lg border-t pt-2 text-pink-600">
                <span>{t.total}:</span>
                <span>{getCartTotal().toLocaleString()} so'm</span>
              </div>
            </div>

            <button 
              onClick={handlePlaceOrder}
              className="w-full bg-pink-600 hover:bg-pink-700 text-white font-bold py-3 rounded-2xl mt-6 transition"
            >
              {t.placeOrder}
            </button>
          </div>
        </div>
      )}
    </div>
  );
}
